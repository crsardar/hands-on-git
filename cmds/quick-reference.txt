# Quick Reference

# See All Users
	git config --global --list

# Set User Name
	git config --global user.name "Chittaranjan Sardar"

# Set E-Mail Add
	git config --global user.email "my_mail@domain.com"

# Status
	git status

# Add 
	git add file_name.txt
	git add .

# Commit 
	git commit -m "Notes for commit"

# Push To Remote 
	git push origin main
	git push
